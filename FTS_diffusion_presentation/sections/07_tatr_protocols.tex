% Constant banner: the message stressed verbally on every slide.
\newcommand{\databanner}{%
  \begin{center}\vspace{-1.5mm}
  \colorbox{NavyBlue!9}{%
    \parbox{0.94\textwidth}{\centering\footnotesize
      \textbf{\color{NavyBlue}Same data budget.}\;
      Every protocol hands the LSTM \emph{exactly} the same
      $Y\times189$ context windows --- only the Markov-chain rollout differs.}}%
  \end{center}}

% One line of the recurring thread strip.
% #1 = this line's index, #2 = how many lines are "evident",
% #3 = accent colour, #4 = protocol name, #5 = description.
\newcommand{\threadrow}[5]{%
  \ifnum#1>#2\relax
    {\color{GraySoft!38}$\bullet$~\texttt{\textbf{#4}}\,---\,#5}%
  \else
    {\color{#3}$\bullet$~\texttt{\textbf{#4}}}~---~#5%
  \fi\par\smallskip}

% The thread strip itself. #1 = number of evident lines (0..3).
\newcommand{\threadstrip}[1]{%
  \begin{center}\vspace{-0.5mm}
  \begin{minipage}{0.94\textwidth}\footnotesize
  \threadrow{1}{#1}{NavyBlue}{authors}%
    {MC \emph{restarts every year} from the \emph{same} state $(p_0,l_0,m_0)$.}
  \threadrow{2}{#1}{Coral}{split}%
    {MC runs \emph{once, continuously}; year edges are only LSTM-window cuts.}
  \threadrow{3}{#1}{TealAcc}{random\_init}%
    {MC restarts every year from a \emph{random} state $\sim$\,\texttt{SEGMENTS\_INIT}.}
  \end{minipage}
  \end{center}}

\note{Set-up slide. Whatever the protocol, the synthetic price series is turned into LSTM training data the same way: a length-64 window (63 days of context, 1-day-ahead target) slides over each 252-day synthetic year, yielding $252-63=189$ context windows per year. For a $Y$-year augmentation that is exactly $Y\times189$ windows. The next three slides change only ONE thing: how the underlying Markov chain produces those 252-day years. The thread strip at the bottom is faded here on purpose --- it lights up protocol by protocol.}

\threadstrip{1}

\note{Protocol used in the paper. Every synthetic year is an INDEPENDENT Markov-chain rollout that starts from the very same conditional state $s_0=(p_0,l_0,m_0)$ --- the chain is rewound to year zero each time. There is no cross-year memory: year 7 looks statistically like year 1. Pro: faithful to the original FTS-Diffusion evaluation. Con: the long-range transition structure of the MC is never exercised. Note the bottom half --- 252 days $\to$ 189 windows $\to$ pool --- which will stay identical on the next two slides.}

Same $s_0$, same $Y\times189$ windows as \texttt{authors}; the only change is that the MC's cross-year transition structure is now actually used.}

\threadstrip{3}

\note{Diagnostic control. Like \texttt{authors}, each year is an independent one-year chain --- but the start state is now drawn at random from \texttt{SEGMENTS\_INIT} (the post-init segments of the real LSTM-train period) instead of the fixed $s_0$. This separates two effects: the deterministic initialisation $(p_0,l_0,m_0)$ versus the MC dynamics. If \texttt{random\_init} matches \texttt{authors}, the fixed init was doing little; if it differs a lot, the init state matters more than the chain itself. All three protocols still feed the LSTM the identical $Y\times189$ windows --- say this out loud.}
\end{frame}
