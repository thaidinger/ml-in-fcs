\begin{columns}[T,onlytextwidth]
\begin{column}{0.50\textwidth}\scriptsize
\textbf{State variables}
\begin{itemize}\setlength\itemsep{0pt}\setlength\topsep{0pt}
  \item $p$: pattern identity (cluster label).
  \item $\alpha$: duration scale factor (length stretch).
  \item $\beta$: magnitude scale factor (amplitude).
\end{itemize}
\vspace{1mm}
\textbf{DTW: align in time before measuring distance.}\\
Two segments stay close even if one is stretched.

\vspace{3mm}

\end{column}

\begin{column}{0.48\textwidth}\scriptsize
\textbf{Why K-means++ for centroid init?}
\begin{itemize}\setlength\itemsep{0pt}\setlength\topsep{0pt}
  \item K-means++ seeds far-apart centroids, more stable clusters.
  \item Plain K-means can collapse two centroids onto the same regime.
\end{itemize}
\end{column}
\end{columns}
\note{Sequential at training and at sampling time. SISC's output is the latent alphabet that conditions PGM. PEM is what stitches isolated motifs into a continuous synthetic series. $\alpha$ and $\beta$ are the per-segment scale factors passed alongside the pattern identity.}
