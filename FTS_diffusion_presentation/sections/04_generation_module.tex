\begin{block}{Takeaway}
\centering
\[
p_\theta(\,\cdot\mid p,\alpha,\beta)\approx q(\,\cdot\mid p,\alpha,\beta)
\]
\scriptsize It is essentially trying to learn the distribution function $q(\,\cdot\mid p,\alpha,\beta)$, where $p$ is the pattern type, $\alpha$ the duration scale, and $\beta$ the magnitude scale.
\end{block}

\begin{frame}{Scaling autoencoder}
\framesubtitle{From variable-length segments to fixed-length latent sequences}
\begin{columns}[onlytextwidth]
\begin{column}{0.48\textwidth}
\begin{alertblock}{Encoder}
Maps a variable-length segment to a padded latent sequence:
\[
x_m \;\longrightarrow\; z_m \in \mathbb{R}^{l_{\max}\times 1}.
\]
\scriptsize\textbf{Released code:}\\
pack input lengths\\
$\to$ 2-layer LSTM, hidden dim $1$\\
$\to$ linear projection, dim $1$
\end{alertblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{alertblock}{Decoder}
Maps the latent sequence back to a variable-length segment:
\[
z_m \;\longrightarrow\; \hat{x}_m.
\]
\scriptsize\textbf{Released code:}\\
pack latent lengths\\
$\to$ 2-layer LSTM, hidden dim $1$\\
$\to$ linear output, dim $1$
\end{alertblock}
\end{column}
\end{columns}

\medskip
\begin{block}{Model choice}
\centering
\small Paper: two LSTM/GRU layers are allowed. Released code: two-layer LSTM encoder and decoder, hidden dimension $1$, padded length $l_{\max}$.
\end{block}
\note{More precise than saying fixed latent vector: the encoder outputs a padded latent sequence of length lmax with one channel. The PGM reshapes this to batch by channel by lmax and subtracts the SISC pattern before diffusion. The decoder packs the latent sequence using the original lengths and maps it back to a univariate segment. The code returns encoder hidden/cell but the decoder forward path does not use them.}
\end{frame}

\begin{frame}{Latent residual for diffusion}
\framesubtitle{The model corrupts the encoded latent sequence relative to the fixed pattern}
\centering
\vspace{-4mm}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.70\textwidth}
\centering
{\scriptsize\bfseries Forward process in latent notation:}\\[-0.5mm]
{\large
\[
q(z^i\mid z^{i-1},p)
=
\mathcal{N}\!\left(
z^i;\sqrt{1-\beta}\,(z^{i-1}-p),\beta I
\right)
\]
}
\end{column}
\begin{column}{0.28\textwidth}
\vspace{2mm}
{\scriptsize
$p$: pattern condition/type\\[1mm]
$\beta$: diffusion noise scale
}
\end{column}
\end{columns}

\vspace{-3mm}
{\small\color{Coral}\bfseries Center at $p$ and diffuse the residual $z^{i-1}-p$.}

\begin{frame}{Deriving the diffusion loss}
\framesubtitle{Train the denoiser by asking it to recover the noise that was added}
\vspace{-2mm}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.31\textwidth}
\begin{block}{1. Residual}
\centering
\[
r_0=z_m-p
\]
\scriptsize Keep $p$ fixed and corrupt the residual.
\end{block}
\end{column}
\begin{column}{0.36\textwidth}
\begin{block}{2. Closed-form noising}
\centering
\[
r_i=\sqrt{\bar{\alpha}_i}r_0+
\sqrt{1-\bar{\alpha}_i}\epsilon
\]
\scriptsize $\epsilon\sim\mathcal{N}(0,I)$, \quad
$\bar{\alpha}_i=\prod_{s=1}^{i}(1-\beta_s)$.
\end{block}
\end{column}
\begin{column}{0.29\textwidth}
\begin{alertblock}{3. Predict noise}
\centering
\[
\epsilon_\theta(r_i,i,p)\approx\epsilon
\]
\scriptsize Train on the known noise sample.
\end{alertblock}
\end{column}
\end{columns}

\vspace{2mm}
\begin{block}{Diffusion loss}
\centering
\[
\mathcal{L}_{\mathrm{diff}}
=
\mathbb{E}_{r_0,i,\epsilon}\!\left[
\|\epsilon-\epsilon_\theta(r_i,i,p)\|_2^2
\right]
\]
\end{block}
\note{This is the DDPM-style derivation behind the paper's epsilon loss. Define the clean residual as $r_0=z_m-p$. The forward process has a closed form, so for a sampled timestep $i$ we can construct $r_i$ from known Gaussian noise $\epsilon$. Since we know the noise that was added, we train the denoiser to predict that noise. This yields the MSE term in Eq. 5.}
\end{frame}

\vspace{1mm}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}\scriptsize
\textbf{Conditional forward step (paper \S4.2):}
\[ q(x^{i}\mid x^{i-1}) = \mathcal{N}\!\bigl(x^{i};\;\sqrt{1-\beta}\,(x^{i-1}-p),\;\beta I\bigr). \]
Noise is added to the residual $(z-p)$, so the denoiser learns to produce segments resembling the centroid of pattern $p$, not arbitrary signals.
\end{column}
\begin{column}{0.43\textwidth}\scriptsize
\begin{itemize}\setlength\itemsep{1pt}\setlength\topsep{0pt}
  \item SAE encoder maps variable-length $\to$ fixed-length latent.
  \item Diffusion runs in the latent, conditioned on $p$.
  \item SAE decoder maps the new latent back to a segment.
  \item \alert{Coherent pattern evolution} prevents learning pure noise.
\end{itemize}
\end{column}
\end{columns}

\begin{frame}{Pattern-conditioned diffusion network}
\framesubtitle{Learning to denoise latent pattern representations}
\begin{block}{Forward diffusion}
Gradually corrupt the latent with Gaussian noise:
\[ q(x^i \mid x^{i-1}) = \mathcal{N}\!\bigl(x^i;\;\sqrt{1-\beta}(x^{i-1}-p),\;\beta I\bigr). \]
\end{block}
\begin{alertblock}{Reverse denoising}
A network learns to recover the previous step, conditional on pattern $p$:
\[ p_{\theta}(x^{i-1} \mid x^i) = \mathcal{N}\!\bigl(x^{i-1};\;\mu_{\theta}(x^i,i,p),\;\beta I\bigr). \]
\end{alertblock}
\begin{block}{Intuition}
Map noise to realistic pattern-level market segments.
\end{block}
\note{The pattern identity $p$ enters as a conditioning variable in both the forward shift and the reverse mean network.}
\end{frame}

\begin{frame}{Training objective: paper vs.\ code}
\framesubtitle{The paper states an unweighted objective; the released code adds weights}
\begin{block}{Paper objective (Eq.\ 5)}
\[
\mathcal{L}(\theta)
=
\mathbb{E}_{x_m}\!\left[\|x_m-\hat{x}_m\|_2^2\right]
+
\mathbb{E}_{x'_m,i,\epsilon}\!\left[
\|\epsilon_i-\epsilon_\theta(x_m^i,i,p)\|_2^2
\right].
\]
\end{block}

\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}\scriptsize
\textbf{Meaning}
\begin{itemize}\setlength\itemsep{0pt}\setlength\topsep{0pt}
  \item First term: reconstruction loss for the scaling autoencoder.
  \item Second term: diffusion noise-prediction loss.
  \item The paper does \alert{not} introduce explicit $\lambda$ weights here.
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}\scriptsize
\begin{alertblock}{Implementation note}
The released code uses:
\[
\texttt{loss\_weights}=[0.98,0.01],\qquad
\texttt{scale\_weight}=0.01.
\]
These are implementation weights, not paper notation.
\end{alertblock}
\end{column}
\end{columns}
\note{Important distinction: Eq. 5 in the paper is unweighted reconstruction plus unweighted diffusion noise prediction. The reference implementation then introduces \texttt{loss\_weights=[0.98,0.01]} and optional \texttt{scale\_weight=0.01} SoftDTW. Present those as code-level choices, not as the paper's mathematical objective.}
\end{frame}
