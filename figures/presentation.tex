% Two figures side-by-side, with mini captions
\newcommand{\twofig}[4]{%
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.5\textwidth}
      \centering\footnotesize\textbf{#2}\\[2pt]
      \adjustbox{max width=\linewidth,max height=0.78\textheight,center}{\includegraphics{#1}}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering\footnotesize\textbf{#4}\\[2pt]
      \adjustbox{max width=\linewidth,max height=0.78\textheight,center}{\includegraphics{#3}}
    \end{column}
  \end{columns}
}

% Three figures side-by-side
\newcommand{\threefig}[6]{%
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.33\textwidth}
      \centering\footnotesize\textbf{#2}\\[2pt]
      \adjustbox{max width=\linewidth,max height=0.78\textheight,center}{\includegraphics{#1}}
    \end{column}
    \begin{column}{0.33\textwidth}
      \centering\footnotesize\textbf{#4}\\[2pt]
      \adjustbox{max width=\linewidth,max height=0.78\textheight,center}{\includegraphics{#3}}
    \end{column}
    \begin{column}{0.33\textwidth}
      \centering\footnotesize\textbf{#6}\\[2pt]
      \adjustbox{max width=\linewidth,max height=0.78\textheight,center}{\includegraphics{#5}}
    \end{column}
  \end{columns}
}

% Single centered figure (with mini caption)
\newcommand{\onefig}[2]{%
  \begin{center}
    \footnotesize\textbf{#2}\\[2pt]
    \adjustbox{max width=0.85\textwidth,max height=0.82\textheight,center}{\includegraphics{#1}}
  \end{center}
}
